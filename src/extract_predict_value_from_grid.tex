\begin{equation}
  \widehat{y_k}(T = t) = \frac{\sum_{l=i-1}^{i+1} \sum_{m=j-1}^{j+1} rain_{l,m(T = t)}}{9}
\end{equation}
\begin{description}
  \item[$\widehat{y_k}$] A prediction value at k-th observation point.
  \item[$T$] Time
  \item[$rain_{l,m}$] The i, j-th prediction value of grid data.
\end{description}
